Create `src/mcp/server.ts`:

```typescript
import { McpServer } from '@modelcontextprotocol/sdk/server/mcp.js';
import { StdioServerTransport } from '@modelcontextprotocol/sdk/server/stdio.js';
import { z } from 'zod';

export async function startMcpServer(): Promise<void> {
  const server = new McpServer({
    name: 'sniff',
    version: '0.1.0',
  });

  // Tool: sniff_scan -- static source analysis (per D-11)
  server.tool(
    'sniff_scan',
    'Run static source code analysis on a project and return findings (placeholder text, TODOs, hardcoded strings, broken links)',
    {
      rootDir: z.string().describe('Absolute path to the project root directory'),
    },
    async ({ rootDir }) => {
      const { loadConfig } = await import('../config/loader.js');
      const { ScannerRegistry } = await import('../scanners/registry.js');
      const { SourceScanner } = await import('../scanners/source/index.js');

      const config = await loadConfig(rootDir);
      const registry = new ScannerRegistry();
      registry.register(new SourceScanner());

      const results = await registry.runAll({ config, rootDir });
      return {
        content: [{ type: 'text' as const, text: JSON.stringify(results, null, 2) }],
      };
    },
  );

  // Tool: sniff_run -- browser E2E + quality scan (per D-11)
  server.tool(
    'sniff_run',
    'Run browser-based E2E quality scan (accessibility, visual regression, performance) against a live URL',
    {
      rootDir: z.string().describe('Absolute path to the project root directory'),
      baseUrl: z.string().describe('Base URL of the running application (e.g., http://localhost:3000)'),
      headless: z.boolean().default(true).describe('Run browser in headless mode'),
    },
    async ({ rootDir, baseUrl, headless }) => {
      const { loadConfig } = await import('../config/loader.js');
      const { BrowserRunner } = await import('../browser/runner.js');
      const { AccessibilityScanner } = await import('../scanners/accessibility/index.js');
      const { VisualRegressionScanner } = await import('../scanners/visual/index.js');
      const { PerformanceScanner } = await import('../scanners/performance/index.js');

      const config = await loadConfig(rootDir);

      // Validate URL
      try { new URL(baseUrl); } catch {
        return { content: [{ type: 'text' as const, text: JSON.stringify({ error: `Invalid URL: ${baseUrl}` }) }] };
      }

      const runner = new BrowserRunner(config);
      runner.registerScanner(new AccessibilityScanner());
      runner.registerScanner(new VisualRegressionScanner());
      runner.registerScanner(new PerformanceScanner());

      const result = await runner.run({
        baseUrl,
        testFiles: ['/'],
        viewports: config.viewports ?? [{ name: 'desktop', width: 1280, height: 720 }],
        headless,
        slowMo: 0,
        timeout: config.browser?.timeout ?? 30000,
      });

      // Type both groups explicitly to avoid Finding vs BrowserFinding incompatibility
      const allFindings: (Finding | BrowserFinding)[] = [
        ...result.scanResults.flatMap(r => r.findings),       // Finding[]
        ...result.pageVisits.flatMap(v => v.findings),         // BrowserFinding[]
      ];
      return {
        content: [{ type: 'text' as const, text: JSON.stringify({
          findings: allFindings,
          urls: result.urls,
          duration: result.duration,
          findingCount: allFindings.length,
        }, null, 2) }],
      };
    },
  );

  // Tool: sniff_report -- load last results (per D-11)
  server.tool(
    'sniff_report',
    'Load the most recent sniff scan results and return a formatted summary',
    {
      rootDir: z.string().describe('Absolute path to the project root directory'),
      format: z.enum(['json', 'summary']).default('summary').describe('Output format: full JSON or text summary'),
    },
    async ({ rootDir, format }) => {
      const { loadLastResults } = await import('../core/persistence.js');
      const results = await loadLastResults(rootDir);

      if (!results) {
        return { content: [{ type: 'text' as const, text: 'No previous sniff results found. Run sniff_scan or sniff_run first.' }] };
      }

      if (format === 'json') {
        return { content: [{ type: 'text' as const, text: JSON.stringify(results, null, 2) }] };
      }

      // Summary format
      const totalFindings = results.results.reduce((sum, r) => sum + r.findings.length, 0);
      const bySeverity: Record<string, number> = {};
      for (const r of results.results) {
        for (const f of r.findings) {
          bySeverity[f.severity] = (bySeverity[f.severity] ?? 0) + 1;
        }
      }
      const scanners = results.results.map(r => r.scanner).join(', ');
      const summary = [
        `Sniff Results (${results.timestamp})`,
        `Scanners: ${scanners}`,
        `Total findings: ${totalFindings}`,
        ...Object.entries(bySeverity).map(([sev, count]) => `  ${sev}: ${count}`),
      ].join('\n');

      return { content: [{ type: 'text' as const, text: summary }] };
    },
  );

  // Connect via stdio transport (per D-12)
  const transport = new StdioServerTransport();
  await server.connect(transport);
}
```

Add first-run Playwright browser check to `src/cli/index.ts`. Add this function and call it before commands that need the browser (`run`, `explore`). Place the function definition at the top of the file after imports:

```typescript
async function ensurePlaywrightBrowsers(): Promise<void> {
  try {
    const { chromium } = await import('playwright');
    chromium.executablePath();
  } catch {
    const pc = (await import('picocolors')).default;
    console.log(pc.yellow('\nPlaywright browsers not installed.'));
    console.log(pc.dim('Installing Chromium (required for browser testing)...\n'));
    const { execSync } = await import('node:child_process');
    try {
      execSync('npx playwright install chromium', { stdio: 'inherit' });
      console.log('');
    } catch {
      console.error(pc.red('Failed to install Playwright browsers. Run manually: npx playwright install chromium'));
      process.exit(1);
    }
  }
}
```

Wire it into the `run` and `explore` command actions (add `await ensurePlaywrightBrowsers();` as the first line inside each action callback, before the lazy import of the command handler). Do NOT add it to `scan`, `report`, `init`, or `ci` commands (they do not need a browser).

## Threat Model

### Trust Boundaries

| Boundary | Description |
|----------|-------------|
| MCP client -> stdio -> MCP server | Untrusted tool invocations from any MCP client (Claude Desktop, Cursor, etc.) |
| MCP tool params -> filesystem/browser | `rootDir` and `baseUrl` parameters control filesystem reads and browser navigation |
| npx -> package registry | Users downloading the package trust npm registry integrity |

### STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation |
|-----------|----------|-----------|-------------|------------|
| T-04-14 | Spoofing | MCP client identity | accept | MCP stdio transport is local-only (no network exposure). The MCP client is the user's own IDE/tool. No authentication needed for local stdio. |
| T-04-15 | Tampering | rootDir parameter | mitigate | Validate `rootDir` is an absolute path and exists on disk before proceeding. Do not allow path traversal patterns. Use `path.resolve()` to normalize. |
| T-04-16 | Tampering | baseUrl parameter | mitigate | Validate `baseUrl` via `new URL()` constructor. Reject non-http/https schemes. Same origin validation as BrowserRunner T-03-01 for subsequent navigations. |
| T-04-17 | Information Disclosure | MCP tool responses | accept | Tool responses contain scan findings and file paths from the user's own project. This is expected behavior -- the user invoked the tool on their own codebase. |
| T-04-18 | Denial of Service | MCP sniff_run | mitigate | Browser timeout enforced via `config.browser.timeout` (default 30s). MCP server does not spawn unbounded processes. Browser closed in BrowserRunner finally block. |
| T-04-19 | Elevation of Privilege | npm publish scope | mitigate | `publishConfig.access: "public"` is explicit. Package uses `files: ["dist"]` to prevent accidental inclusion of `.env`, credentials, or source files in the published package. |
</threat_model>
