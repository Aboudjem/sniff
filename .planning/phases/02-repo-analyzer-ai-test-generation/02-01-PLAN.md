```typescript
export interface FrameworkInfo {
  name: 'nextjs' | 'react' | 'vue' | 'svelte' | 'unknown';
  version?: string;
  configFiles: string[];
}

export interface RouteInfo {
  path: string;           // e.g., "/dashboard/settings"
  filePath: string;       // e.g., "src/app/dashboard/settings/page.tsx"
  framework: string;      // e.g., "nextjs-app"
  dynamic: boolean;       // true if route has dynamic segments
  params?: string[];      // ["id"] for /users/[id]
}

export interface ElementInfo {
  tag: string;            // "button", "input", "a", "form", "select", "textarea"
  testId?: string;        // data-testid value
  id?: string;
  name?: string;
  ariaLabel?: string;
  role?: string;
  text?: string;          // static text content
  type?: string;          // input type
  href?: string;          // link href
  filePath: string;
  line: number;
}

export interface ComponentInfo {
  name: string;
  filePath: string;
  exports: string[];
  hasDefaultExport: boolean;
  elements: ElementInfo[];
  routes: string[];
}

export interface AnalysisResult {
  project: {
    name: string;
    frameworks: FrameworkInfo[];
    rootDir: string;
  };
  routes: RouteInfo[];
  components: ComponentInfo[];
  elements: ElementInfo[];
  metadata: {
    analyzedAt: string;
    duration: number;
    fileCount: number;
    routeCount: number;
    elementCount: number;
  };
}
```

Create `src/analyzers/route-discoverer.ts` — export async function `discoverRoutes(rootDir: string, frameworks: FrameworkInfo[]): Promise<RouteInfo[]>`.

Implementation per D-02, D-04, RESEARCH.md Section 2:

1. **Next.js App Router** — if framework is `nextjs`:
   - Check if `app/` directory exists at rootDir (or `src/app/`)
   - Use `fast-glob` to find `**/page.{tsx,jsx,ts,js}` under app directory
   - Convert file path to route: strip `app/` prefix, strip `/page.tsx` suffix, strip route groups `(groupname)/`, convert `[param]` segments
   - Set `dynamic: true` and extract `params` array for `[param]` segments
   - Set `framework: 'nextjs-app'`
   - Exclude `layout.tsx`, `loading.tsx`, `error.tsx`, `not-found.tsx` (only `page.*` files are routes)

2. **Next.js Pages Router** — if framework is `nextjs`:
   - Check if `pages/` directory exists at rootDir (or `src/pages/`)
   - Use `fast-glob` to find `**/*.{tsx,jsx,ts,js}` under pages directory
   - Exclude `_app`, `_document`, `_error` files
   - Convert file path to route: strip `pages/` prefix, strip extension, `index` -> `/`
   - Set `framework: 'nextjs-pages'`

3. **SvelteKit** — if framework is `svelte`:
   - Check if `src/routes/` directory exists
   - Find `**/ +page.svelte` files
   - Convert similarly to Next.js App Router: strip `src/routes/` prefix, strip `/+page.svelte`
   - Set `framework: 'sveltekit'`

4. **React Router (basic)** — if framework is `react`:
   - Use `fast-glob` to find source files, then `readFile` + regex to find `createBrowserRouter` or `<Route path=` patterns
   - Extract `path` values via regex: `/path:\s*['"]([^'"]+)['"]/g` and `/path=['"]([^'"]+)['"]/g`
   - Set `framework: 'react-router'`, `dynamic` based on `:param` presence
   - Flag these as approximate (static analysis may miss dynamic routes)

5. **Vue Router (basic)** — if framework is `vue`:
   - Look for `router/index.{ts,js}` or `router.{ts,js}`
   - Extract `path` from route config arrays via regex
   - Set `framework: 'vue-router'`

6. Return combined `RouteInfo[]` from all detected frameworks. Use `relative()` for filePath values.
