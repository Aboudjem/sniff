Implementation per D-05, D-06, RESEARCH.md Section 3:

- For each file in filePaths:
  - Read file content with `readFile(filePath, 'utf-8')`
  - Parse with `@babel/parser` using plugins: `['typescript', 'jsx']` and `sourceType: 'module'`
  - Traverse AST with `@babel/traverse` using `JSXOpeningElement` visitor
  - For each JSX opening element, check if tag name is one of: `form`, `button`, `a`, `input`, `select`, `textarea`
  - If interactive, extract attributes:
    - `data-testid` -> testId
    - `id` -> id
    - `name` -> name
    - `aria-label` -> ariaLabel
    - `role` -> role
    - `type` -> type (for input)
    - `href` -> href (for a)
  - Extract static text content from JSXText children (for buttons/links)
  - Record line number from AST node `loc.start.line`
  - Set `filePath: relative(rootDir, absoluteFilePath)`
  - Build ComponentInfo per file: detect component name from default export or named export (look for `ExportDefaultDeclaration` and `ExportNamedDeclaration` visitors), collect elements belonging to each component

- Process files in batches of 50 (per existing codebase pattern)
- Wrap individual file parsing in try/catch — on parse error, skip file and continue (graceful degradation)
- Return combined `{ elements, components }`

Helper to extract JSX attribute value:
```typescript
function getJSXAttrValue(attr: JSXAttribute): string | undefined {
  if (!attr.value) return undefined;
  if (attr.value.type === 'StringLiteral') return attr.value.value;
  if (attr.value.type === 'JSXExpressionContainer' && attr.value.expression.type === 'StringLiteral') {
    return attr.value.expression.value;
  }
  return undefined;
}
```

2. Create `src/scanners/repo-analyzer.ts` — follow `SourceScanner` pattern exactly:

```typescript
import { performance } from 'node:perf_hooks';
import type { Scanner, ScanContext, ScanResult } from './types.js';

export class RepoAnalyzer implements Scanner {
  name = 'repo-analyzer';

  async scan(ctx: ScanContext): Promise<ScanResult> {
    const start = performance.now();

    // Lazy-load analyzer modules
    const { detectFrameworks } = await import('../analyzers/framework-detector.js');
    const { discoverRoutes } = await import('../analyzers/route-discoverer.js');
    const { extractElements } = await import('../analyzers/element-extractor.js');

    // 1. Detect frameworks
    const frameworks = await detectFrameworks(ctx.rootDir);

    // 2. Discover routes
    const routes = await discoverRoutes(ctx.rootDir, frameworks);

    // 3. Extract elements from route files
    const routeFilePaths = routes
      .map(r => join(ctx.rootDir, r.filePath))
      .filter(f => f.endsWith('.tsx') || f.endsWith('.jsx') || f.endsWith('.ts') || f.endsWith('.js'));
    const { elements, components } = await extractElements(routeFilePaths, ctx.rootDir);

    // 4. Build AnalysisResult
    const analysis = {
      project: {
        name: basename(ctx.rootDir),
        frameworks,
        rootDir: ctx.rootDir,
      },
      routes,
      components,
      elements,
      metadata: {
        analyzedAt: new Date().toISOString(),
        duration: performance.now() - start,
        fileCount: routeFilePaths.length,
        routeCount: routes.length,
        elementCount: elements.length,
      },
    };

    return {
      scanner: this.name,
      findings: [], // Analyzer produces analysis, not findings
      duration: performance.now() - start,
      metadata: { analysis },
    };
  }
}
```

Use `node:path` for `join` and `basename`. Follow lazy-import pattern for all analyzer modules.
